% MCQ Styling - Internal Version (Squares)

% Box styling for questions
\usepackage{tcolorbox}
\tcbuselibrary{skins,breakable}
\usepackage{amssymb} % For special symbols

% Define MCQ box style - make it non-breakable and more compact
\newtcolorbox{mcqbox}{
  enhanced,
  breakable=false, % Prevent breaking across pages
  colback=white,
  colframe=questblue,
  arc=5mm,
  boxrule=0.5mm,
  leftrule=2mm,
  title={\textbf{Question \thetotalcounter}},
  fonttitle=\bfseries\normalsize,
  coltitle=white,
  colbacktitle=questblue,
  attach boxed title to top left={yshift=-1mm, xshift=2mm},
  boxed title style={rounded corners, boxrule=0pt},
  before={\par\vspace{0cm}}, % No extra spacing before
  after={\par\vspace{0.05cm}},  % Minimal spacing after
}

% Option styling - using same symbol for all options
\newcommand{\correctoption}[1]{%
  \item[\textcolor{questblue}{$\blacksquare$} \alph{enumi}] #1
}

\newcommand{\incorrectoption}[1]{%
  \item[\textcolor{questblue}{$\blacksquare$} \alph{enumi}] #1
}

% Define enhanced MCQ environment - more compact with no page breaks
\newenvironment{enhancedmcq}[2][]
{%
  \stepcounter{questioncounter}%
  \begin{mcqbox}%
  \textbf{#2}\vspace{-0.15cm}%
  \begin{enumerate}[label=\alph*.,itemsep=0.03cm,parsep=0pt,leftmargin=1.5cm]% Reduced spacing
}
{%
  \end{enumerate}%
  \end{mcqbox}%
  \stepcounter{totalcounter}%
  \vspace{0.1cm}% Reduced spacing
}
